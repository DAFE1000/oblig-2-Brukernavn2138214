import numpy as np
import matplotlib.pyplot as plt

def f(x):
    return np.exp(-x/4) * np.arctan(x)

x = np.linspace(0, 3, 1000)
y = f(x)

index = np.argmax(y)
x_max = x[index]
y_max = y[index]

print("x ≈", round(x_max, 4))
print("f(x) ≈", round(y_max, 4))

plt.plot(x, y)
plt.scatter(x_max, y_max)
plt.show()
